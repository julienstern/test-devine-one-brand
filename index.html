    <div class="container">

            <p class="subtitle">Devine le nouveau nom avant l'annonce officielle</p>

                ⏰ Chargement...

            <div class="input-group">

                    <span class="predictions-title">📋 Tous les pronostics</span>

                    <span class="count-badge" id="countBadge">0 pronostic</span>

                </div>

                <div class="predictions-list" id="predictionsList">

    <script>

        // Date de l'annonce : mardi 26 mai 2025

        const revealDate = new Date('2025-05-26T09:00:00');


        // Charger les pronostics depuis localStorage

        function loadPredictions() {

            const stored = localStorage.getItem('zeplug_predictions');

            return stored ? JSON.parse(stored) : [];

        }


        // Sauvegarder les pronostics

        function savePredictions(predictions) {

            localStorage.setItem('zeplug_predictions', JSON.stringify(predictions));

        }


        // Afficher les pronostics

        function renderPredictions() {

            const predictions = loadPredictions();

            const list = document.getElementById('predictionsList');

            const badge = document.getElementById('countBadge');


            badge.textContent = `${predictions.length} pronostic${predictions.length > 1 ? 's' : ''}`;


            if (predictions.length === 0) {

                list.innerHTML = '<div class="empty-state">Aucun pronostic pour le moment. Sois le premier ! 🚀</div>';

                return;

            }


            list.innerHTML = predictions.map(p => `

                <div class="prediction-item">

                    <span class="prediction-name">${escapeHtml(p.name)}</span>

                    <div class="prediction-meta">

                        <div class="prediction-author">${escapeHtml(p.author)}</div>

                        <div class="prediction-time">${p.time}</div>

                    </div>

                </div>

            `).join('');

        }


        // Échapper le HTML pour éviter les injections

        function escapeHtml(text) {

            const div = document.createElement('div');

            div.textContent = text;

            return div.innerHTML;

        }


        // Soumettre un pronostic

        function submitPrediction() {

            const authorInput = document.getElementById('authorInput');

            const predictionInput = document.getElementById('predictionInput');


            const author = authorInput.value.trim();

            const name = predictionInput.value.trim();


            if (!author) {

                alert('Entre ton prénom !');

                authorInput.focus();

                return;

            }


            if (!name) {

                alert('Entre ta proposition de nom !');

                predictionInput.focus();

                return;

            }


            const predictions = loadPredictions();

            

            predictions.unshift({

                author: author,

                name: name,

                time: new Date().toLocaleString('fr-FR', {

                    day: 'numeric',

                    month: 'short',

                    hour: '2-digit',

                    minute: '2-digit'

                })

            });


            savePredictions(predictions);

            renderPredictions();


            predictionInput.value = '';

            

            // Petit feedback visuel

            const btn = document.querySelector('button');

            btn.textContent = '✅ Ajouté !';

            setTimeout(() => {

                btn.textContent = '🎯 Valider mon pronostic';

            }, 1500);

        }


        // Réinitialiser tous les pronostics

        function clearAll() {

            if (confirm('Supprimer tous les pronostics ?')) {

                localStorage.removeItem('zeplug_predictions');

                renderPredictions();

            }

        }


        // Compte à rebours

        function updateCountdown() {

            const now = new Date();

            const diff = revealDate - now;


            if (diff <= 0) {

                document.getElementById('countdown').textContent = '🎉 L\'annonce est faite !';

                return;

            }


            const days = Math.floor(diff / (1000 * 60 * 60 * 24));

            const hours = Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));

            const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));


            let text = '⏰ ';

            if (days > 0) text += `${days}j `;

            text += `${hours}h ${minutes}min avant l'annonce`;


            document.getElementById('countdown').textContent = text;

        }


        // Permettre la validation avec Entrée

        document.getElementById('predictionInput').addEventListener('keypress', function(e) {

            if (e.key === 'Enter') submitPrediction();

        });


        // Initialisation

        renderPredictions();

        updateCountdown();

        setInterval(updateCountdown, 60000);

    </script>
